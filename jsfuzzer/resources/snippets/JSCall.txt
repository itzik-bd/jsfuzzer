var JSCallDepth = -1;
function JSCall() {
	var f = arguments['0'];
	if (typeof f === 'function') {
		var params = [];
		for (key in arguments) if (key != '0') params.push(arguments[key]);
		JSCallDepth++;
		var result = f.apply(f, params);
		JSCallDepth--;
		return result;
	}
}
function JSCallDebug(name) {
	JSPrint(Array(JSCallDepth+1).join(" ") + name);
}
